================================================================================
                    المرحلة 5: تتبع حركات المخزون (Inventory Movements)
================================================================================

📋 نظرة عامة:
--------------
هذا الجدول هو السجل الموحد لكل حركة تحدث في المخزون من أي مصدر.
كل عملية تُغيّر المخزون (بيع / شراء / فاقد / مرتجع / تعديل) تُسجَّل هنا تلقائياً.
✅ يُجيب على سؤال واحد: "ما الذي حدث للمخزون؟ متى؟ ولماذا؟"

================================================================================
                        🗂️ الجدول المستخدم (1 جدول)
================================================================================

┌─────────────────────────────────────────────────────────────────────────────┐
│ 1️⃣ inventory_movements                                                      │
└─────────────────────────────────────────────────────────────────────────────┘

📌 الغرض: سجل موحد لكل حركات المخزون — يتغذى تلقائياً من جميع العمليات

الحقل              النوع          العلاقة             القيمة الافتراضية    الوصف
──────────────────────────────────────────────────────────────────────────────────
id                 BIGINT         PK                  AUTO_INCREMENT        المعرف
product_id         BIGINT         FK → products.id    —                     المنتج
movement_type      ENUM           —                   —                     نوع الحركة
quantity           DECIMAL(10,2)  —                   —                     الكمية (موجب/سالب)
quantity_before    DECIMAL(10,2)  —                   —                     الكمية قبل الحركة
quantity_after     DECIMAL(10,2)  —                   —                     الكمية بعد الحركة
reference_type     VARCHAR(50)    —                   NULL                  نوع المرجع
reference_id       BIGINT         —                   NULL                  معرف المرجع
user_id            BIGINT         FK → users.id       —                     المستخدم المنفذ
notes              TEXT           —                   NULL                  ملاحظات
created_at         TIMESTAMP      —                   CURRENT_TIMESTAMP     تاريخ الحركة

ENUM movement_type:
   'in'          → إضافة مخزون
   'out'         → خصم مخزون
   'adjustment'  → تعديل يدوي

reference_type القيم الممكنة:
   'invoice_item' → مرتبط بـ invoice_items.id (بيع)
   'purchase'     → مرتبط بـ purchases.id (شراء)
   'return'       → مرتبط بـ returns.id (مرتجع)
   NULL           → تعديل يدوي بدون مرجع

Constraints:
   INDEX (product_id)
   INDEX (movement_type)
   INDEX (reference_type, reference_id)
   INDEX (created_at)

================================================================================
                        🔄 مخطط تدفق الحركات
================================================================================

   بيع (المرحلة 3)        → inventory_movements (out, ref: invoice_item)
   شراء (المرحلة 4)       → inventory_movements (in,  ref: purchase)
   مرتجع (المرحلة 6)      → inventory_movements (in,  ref: return)
   فاقد/تلف (يدوي)        → inventory_movements (out, ref: NULL)
   تعديل يدوي (جرد)       → inventory_movements (adjustment, ref: NULL)
                                    ↓
                          inventory.quantity (يُحدَّث دائماً)

================================================================================
                        📊 استعلامات مفيدة
================================================================================

┌─────────────────────────────────────────────────────────────────────────────┐
│ 1️⃣ عرض تاريخ حركات منتج معين                                              │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    im.created_at,
    im.movement_type,
    im.quantity,
    im.quantity_before,
    im.quantity_after,
    im.reference_type,
    im.notes,
    u.name AS user_name
FROM inventory_movements im
JOIN users u ON im.user_id = u.id
WHERE im.product_id = 1
ORDER BY im.created_at DESC;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 2️⃣ إجمالي المبيعات من المخزون (حسب المنتج)                               │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    p.name,
    SUM(ABS(im.quantity)) AS total_sold
FROM inventory_movements im
JOIN products p ON im.product_id = p.id
WHERE im.movement_type = 'out'
  AND im.reference_type = 'invoice_item'
GROUP BY p.id, p.name
ORDER BY total_sold DESC;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 3️⃣ إجمالي المشتريات (حسب المنتج)                                         │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    p.name,
    SUM(im.quantity) AS total_purchased
FROM inventory_movements im
JOIN products p ON im.product_id = p.id
WHERE im.movement_type = 'in'
  AND im.reference_type = 'purchase'
GROUP BY p.id, p.name
ORDER BY total_purchased DESC;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 4️⃣ إجمالي الفاقد والتلف                                                   │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    p.name,
    SUM(ABS(im.quantity)) AS total_loss
FROM inventory_movements im
JOIN products p ON im.product_id = p.id
WHERE im.movement_type = 'out'
  AND im.reference_type IS NULL
GROUP BY p.id, p.name
ORDER BY total_loss DESC;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 5️⃣ التحقق من صحة المخزون (Inventory Audit)                                │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    p.name,
    i.quantity AS current_inventory,
    (
        SELECT COALESCE(SUM(quantity), 0)
        FROM inventory_movements
        WHERE product_id = p.id
    ) AS calculated_inventory,
    (i.quantity - (
        SELECT COALESCE(SUM(quantity), 0)
        FROM inventory_movements
        WHERE product_id = p.id
    )) AS difference
FROM products p
JOIN inventory i ON p.id = i.product_id;

-- إذا كان difference = 0 → المخزون صحيح ✅
-- إذا كان difference ≠ 0 → يوجد خطأ ❌


┌─────────────────────────────────────────────────────────────────────────────┐
│ 6️⃣ حركات المخزون خلال فترة معينة                                          │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    p.name,
    im.movement_type,
    SUM(im.quantity) AS total_quantity
FROM inventory_movements im
JOIN products p ON im.product_id = p.id
WHERE im.created_at BETWEEN '2024-01-01' AND '2024-01-31'
GROUP BY p.id, p.name, im.movement_type
ORDER BY p.name, im.movement_type;
