📌 الغرض: تعريف الخصومات والعروض الترويجية

الحقل              النوع          القيمة الافتراضية    الوصف
──────────────────────────────────────────────────────────────────────────────
id                 BIGINT         AUTO_INCREMENT        المعرف
name               VARCHAR(255)   —                     اسم الخصم / العرض
type               ENUM           —                     نوع الخصم
value              DECIMAL(10,2)  —                     قيمة الخصم
min_purchase       DECIMAL(10,2)  NULL                  الحد الأدنى للشراء (اختياري)
start_date         DATE           NULL                  تاريخ بداية العرض
end_date           DATE           NULL                  تاريخ نهاية العرض
is_active          BOOLEAN        TRUE                  حالة النشاط
created_at         TIMESTAMP      CURRENT_TIMESTAMP     تاريخ الإنشاء
updated_at         TIMESTAMP      CURRENT_TIMESTAMP     تاريخ التحديث

ENUM type:
   'percentage'  → خصم نسبة مئوية (مثال: 10%)
   'fixed'       → خصم مبلغ ثابت (مثال: 5 دينار)

Constraints:
   INDEX (is_active)
   INDEX (start_date, end_date)

📌 أمثلة:
   خصم 10% على جميع المنتجات
   خصم 5 دينار على فاتورة فوق 100 دينار
   خصم 15% على Lacoste White فقط

📌 الغرض: ربط الخصومات بالمنتجات المحددة
📌 ملاحظة: خصم بدون ربط = يُطبَّق على جميع المنتجات

الحقل              النوع      العلاقة                القيمة الافتراضية    الوصف
──────────────────────────────────────────────────────────────────────────────────
id                 BIGINT     PK                     AUTO_INCREMENT        المعرف
discount_id        BIGINT     FK → discounts.id      —                     الخصم
product_id         BIGINT     FK → products.id       —                     المنتج
created_at         TIMESTAMP  —                      CURRENT_TIMESTAMP     تاريخ الإنشاء

Constraints:
   UNIQUE (discount_id, product_id)
   INDEX (discount_id)
   INDEX (product_id)

┌─────────────────────────────────────────────────────────────────────────────┐
│ الخطوة 3: تطبيق الخصم عند البيع (تلقائي)                                 │
└─────────────────────────────────────────────────────────────────────────────┘

🔹 الفاعل: النظام (Auto — عند إنشاء البيع)

📊 الجداول المتأثرة:
   - discounts (قراءة)
   - discount_products (قراءة)
   - invoice_items (كتابة - discount)
   - invoices (كتابة - discount)

🔧 العملية:

   ┌─────────────────────────────────────────────────────────────────────────┐
   │ أ. البحث عن خصم نشط للمنتج                                            │
   └─────────────────────────────────────────────────────────────────────────┘
   SELECT d.*
   FROM discounts d
   LEFT JOIN discount_products dp ON d.id = dp.discount_id
   WHERE d.is_active = TRUE
     AND (dp.product_id = ? OR dp.product_id IS NULL)
     AND (d.start_date IS NULL OR d.start_date <= CURDATE())
     AND (d.end_date IS NULL OR d.end_date >= CURDATE())
   LIMIT 1;

   ┌─────────────────────────────────────────────────────────────────────────┐
   │ ب. حساب قيمة الخصم                                                     │
   └─────────────────────────────────────────────────────────────────────────┘
   IF type = 'percentage':
       discount_amount = unit_price × (value / 100)
   ELSE IF type = 'fixed':
       discount_amount = value

   ┌─────────────────────────────────────────────────────────────────────────┐
   │ ج. تطبيق الخصم على السطر                                              │
   └─────────────────────────────────────────────────────────────────────────┘
   INSERT INTO invoice_items (invoice_id, product_id, sale_type, size_id,
                               quantity, unit_price, discount, total_price)
   VALUES (?, ?, ?, ?, ?, ?, discount_amount, unit_price - discount_amount);

   ┌─────────────────────────────────────────────────────────────────────────┐
   │ د. تطبيق خصم على مستوى الفاتورة (إذا وُجد)                            │
   └─────────────────────────────────────────────────────────────────────────┘
   -- البحث عن خصم على مستوى الفاتورة
   SELECT d.*
   FROM discounts d
   WHERE d.is_active = TRUE
     AND d.min_purchase IS NOT NULL
     AND d.min_purchase <= invoice_subtotal
     AND NOT EXISTS (SELECT 1 FROM discount_products WHERE discount_id = d.id)
   LIMIT 1;

   -- تطبيق الخصم
   UPDATE invoices
   SET discount = ?,
       total = subtotal - ?
   WHERE id = ?;


================================================================================
                        🔄 منطق تطبيق الخصم
================================================================================

   اختيار المنتج
         ↓
   البحث عن خصم نشط:
   ┌─────────────────────────────────────────────────────────┐
   │ 1. is_active = TRUE                                     │
   │ 2. (product_id = X OR لا يوجد ربط بمنتجات)            │
   │ 3. (start_date <= اليوم <= end_date OR لا يوجد تاريخ) │
   └─────────────────────────────────────────────────────────┘
         ↓
   ┌─────────────────────────────────────────────────────────┐
   │ type = 'percentage': discount = price × (value/100)     │
   │ type = 'fixed':      discount = value                   │
   └─────────────────────────────────────────────────────────┘
         ↓
   السعر النهائي = السعر الأصلي - discount

INSERT INTO discounts (name, type, value, min_purchase, is_active)
VALUES ('خصم 5 دينار على فاتورة فوق 100', 'fixed', 5, 100, TRUE);

================================================================================
                        📊 استعلامات مفيدة
================================================================================

┌─────────────────────────────────────────────────────────────────────────────┐
│ 1️⃣ عرض جميع الخصومات النشطة                                               │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    d.id,
    d.name,
    d.type,
    d.value,
    d.min_purchase,
    d.start_date,
    d.end_date,
    COUNT(dp.product_id) AS products_count
FROM discounts d
LEFT JOIN discount_products dp ON d.id = dp.discount_id
WHERE d.is_active = TRUE
  AND (d.start_date IS NULL OR d.start_date <= CURDATE())
  AND (d.end_date IS NULL OR d.end_date >= CURDATE())
GROUP BY d.id;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 2️⃣ عرض المنتجات المشمولة بخصم معين                                        │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    p.id,
    p.name,
    d.name AS discount_name,
    d.type,
    d.value
FROM discount_products dp
JOIN products p ON dp.product_id = p.id
JOIN discounts d ON dp.discount_id = d.id
WHERE d.id = 1;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 3️⃣ إجمالي الخصومات الممنوحة                                               │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    SUM(ii.discount) AS total_item_discounts,
    SUM(i.discount) AS total_invoice_discounts,
    SUM(ii.discount) + SUM(i.discount) AS total_discounts
FROM invoices i
LEFT JOIN invoice_items ii ON i.id = ii.invoice_id
WHERE i.created_at BETWEEN '2024-01-01' AND '2024-01-31';


┌─────────────────────────────────────────────────────────────────────────────┐
│ 4️⃣ أكثر الخصومات استخداماً                                                │
└─────────────────────────────────────────────────────────────────────────────┘
-- هذا يحتاج إضافة حقل discount_id في invoice_items (اختياري)
-- أو يمكن تتبعه من خلال notes


┌─────────────────────────────────────────────────────────────────────────────┐
│ 5️⃣ تأثير الخصومات على الإيرادات                                          │
└─────────────────────────────────────────────────────────────────────────────┘
SELECT 
    DATE(i.created_at) AS sale_date,
    SUM(i.subtotal) AS revenue_before_discount,
    SUM(i.discount) + SUM(ii.discount) AS total_discounts,
    SUM(i.total) AS revenue_after_discount,
    ROUND((SUM(i.discount) + SUM(ii.discount)) / SUM(i.subtotal) * 100, 2) AS discount_percentage
FROM invoices i
LEFT JOIN invoice_items ii ON i.id = ii.invoice_id
WHERE i.created_at BETWEEN '2024-01-01' AND '2024-01-31'
GROUP BY DATE(i.created_at)
ORDER BY sale_date DESC;

================================================================================
                        📊 التقارير الجديدة
================================================================================

┌─────────────────────────────────────────────────────────────────────────────┐
│ 1️⃣ أكثر الخصومات استخداماً                                                │
└─────────────────────────────────────────────────────────────────────────────┘

SELECT 
    d.id,
    d.name,
    d.type,
    d.value,
    COUNT(DISTINCT ii.invoice_id) as times_used,
    SUM(ii.discount) as total_discount_amount,
    SUM(ii.total_price) as total_sales_with_discount
FROM invoice_items ii
JOIN discounts d ON ii.discount_id = d.id
WHERE ii.discount_id IS NOT NULL
GROUP BY d.id, d.name, d.type, d.value
ORDER BY times_used DESC;

✅ النتيجة:
| id | name                    | type       | value | times_used | total_discount | total_sales |
|----|-------------------------|------------|-------|------------|----------------|-------------|
| 5  | خصم 10% على Lacoste    | percentage | 10    | 150        | 600            | 5400        |
| 3  | خصم 5 دينار فوق 100    | fixed      | 5     | 80         | 400            | 9600        |


┌─────────────────────────────────────────────────────────────────────────────┐
│ 2️⃣ فعالية الخصومات (ROI)                                                  │
└─────────────────────────────────────────────────────────────────────────────┘

SELECT 
    d.name,
    COUNT(DISTINCT ii.invoice_id) as invoices_count,
    SUM(ii.discount) as total_discount_given,
    SUM(ii.total_price) as total_revenue,
    ROUND(SUM(ii.total_price) / SUM(ii.discount), 2) as revenue_per_discount_dinar
FROM invoice_items ii
JOIN discounts d ON ii.discount_id = d.id
WHERE ii.discount_id IS NOT NULL
GROUP BY d.id, d.name
ORDER BY revenue_per_discount_dinar DESC;

✅ الفائدة: معرفة أي خصم يحقق أعلى عائد

│ 3️⃣ الخصومات حسب الفترة الزمنية                                           │
└─────────────────────────────────────────────────────────────────────────────┘

SELECT 
    DATE(i.created_at) as sale_date,
    d.name as discount_name,
    COUNT(*) as usage_count,
    SUM(ii.discount) as total_discount
FROM invoice_items ii
JOIN invoices i ON ii.invoice_id = i.id
JOIN discounts d ON ii.discount_id = d.id
WHERE i.created_at BETWEEN '2024-01-01' AND '2024-01-31'
  AND ii.discount_id IS NOT NULL
GROUP BY DATE(i.created_at), d.id, d.name
ORDER BY sale_date DESC, usage_count DESC;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 4️⃣ المنتجات الأكثر استفادة من الخصومات                                   │
└─────────────────────────────────────────────────────────────────────────────┘

SELECT 
    p.name as product_name,
    d.name as discount_name,
    COUNT(*) as times_discounted,
    AVG(ii.discount) as avg_discount,
    SUM(ii.total_price) as total_sales
FROM invoice_items ii
JOIN products p ON ii.product_id = p.id
JOIN discounts d ON ii.discount_id = d.id
WHERE ii.discount_id IS NOT NULL
GROUP BY p.id, p.name, d.id, d.name
ORDER BY times_discounted DESC
LIMIT 10;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 5️⃣ مقارنة المبيعات مع وبدون خصومات                                       │
└─────────────────────────────────────────────────────────────────────────────┘

SELECT 
    'مع خصم' as category,
    COUNT(*) as items_count,
    SUM(unit_price * quantity) as original_amount,
    SUM(discount) as discount_amount,
    SUM(total_price) as final_amount
FROM invoice_items
WHERE discount_id IS NOT NULL

UNION ALL

SELECT 
    'بدون خصم' as category,
    COUNT(*) as items_count,
    SUM(unit_price * quantity) as original_amount,
    0 as discount_amount,
    SUM(total_price) as final_amount
FROM invoice_items
WHERE discount_id IS NULL;


┌─────────────────────────────────────────────────────────────────────────────┐
│ 6️⃣ تحليل سلوك العملاء مع الخصومات                                        │
└─────────────────────────────────────────────────────────────────────────────┘

SELECT 
    c.name as customer_name,
    COUNT(DISTINCT i.id) as total_invoices,
    COUNT(DISTINCT CASE WHEN ii.discount_id IS NOT NULL THEN i.id END) as invoices_with_discount,
    ROUND(COUNT(DISTINCT CASE WHEN ii.discount_id IS NOT NULL THEN i.id END) * 100.0 / COUNT(DISTINCT i.id), 2) as discount_usage_percentage,
    SUM(CASE WHEN ii.discount_id IS NOT NULL THEN ii.discount ELSE 0 END) as total_discounts_received
FROM customers c
JOIN invoices i ON c.id = i.customer_id
LEFT JOIN invoice_items ii ON i.id = ii.invoice_id
GROUP BY c.id, c.name
HAVING total_invoices > 5
ORDER BY discount_usage_percentage DESC;
